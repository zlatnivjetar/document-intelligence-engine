**Rule 1: `files` array must include only `dist`**
Correct value: `"files": ["dist"]`
Do NOT include `src`, `*.ts`, or `node_modules`.

**Rule 2: `exports` must include a `types` condition BEFORE `import`**
The Node.js resolver checks conditions in declaration order. Types must come first:
```json
"exports": {
  ".": {
    "types": "./dist/index.d.mts",
    "import": "./dist/index.mjs",
    "require": "./dist/index.cjs"
  }
}
```

**Rule 3: `peerDependencies` — only `zod` and `ai`**
Consumers provide their own `zod` and `ai`. They should NOT need to provide `@ai-sdk/anthropic`, `@ai-sdk/provider`, or `@anthropic-ai/sdk` — these are DocPipe implementation details.

If `@ai-sdk/anthropic` or `@anthropic-ai/sdk` appear in `peerDependencies`, remove them (they belong in `dependencies` only).

**Rule 4: `peerDependenciesMeta` — mark peers as optional**
Since consumers CAN use DocPipe with a different provider by passing their own `LanguageModelV3`:
```json
"peerDependenciesMeta": {
  "ai": { "optional": true },
  "zod": { "optional": false }
}
```

**Rule 5: `engines` field — document Node.js requirement**
```json
"engines": { "node": ">=18.0.0" }
```

**Rule 6: Verify `main`, `module`, `types` top-level fields point to correct dist files**
- `"main": "./dist/index.cjs"` (CJS for older tooling)
- `"module": "./dist/index.mjs"` (ESM hint for bundlers)

```javascript
#!/usr/bin/env node
// scripts/verify-consumer.mjs
// Validates that @docpipe/core is installable and importable outside the monorepo.
import { execSync } from 'child_process';
import { mkdtempSync, writeFileSync, rmSync } from 'fs';
import { tmpdir } from 'os';
import { join, resolve } from 'path';
import { fileURLToPath } from 'url';

const __dirname = fileURLToPath(new URL('.', import.meta.url));
const repoRoot = resolve(__dirname, '..');
const coreDir = join(repoRoot, 'packages', 'core');

function run(cmd, cwd = repoRoot) {
  console.log(`$ ${cmd}`);
  return execSync(cmd, { cwd, stdio: 'pipe', encoding: 'utf8' });
}

let tempDir;
try {
  // 1. Build fresh dist
  run('pnpm --filter @docpipe/core build');

  // 2. Pack the tarball
  const packOutput = run('npm pack --json', coreDir);
  const [{ filename }] = JSON.parse(packOutput);
  const tarball = join(coreDir, filename);
  console.log(`Packed: ${tarball}`);

  // 3. Create temp consumer directory
  tempDir = mkdtempSync(join(tmpdir(), 'docpipe-consumer-test-'));
  console.log(`Consumer dir: ${tempDir}`);

  // 4. Init package.json
  writeFileSync(join(tempDir, 'package.json'), JSON.stringify({
    name: 'docpipe-consumer-test',
    version: '1.0.0',
    type: 'module',
    private: true,
  }, null, 2));

  // 5. Install tarball + peers
  run(`npm install "${tarball}" ai@"^6" zod@"^4" @ai-sdk/anthropic@"^3" --no-save`, tempDir);

  // 6. Write consumer test
  const testScript = `
import { extract, createAnthropicProvider, invoiceSchema } from '@docpipe/core';

if (typeof extract !== 'function') throw new Error('extract is not a function');
if (typeof createAnthropicProvider !== 'function') throw new Error('createAnthropicProvider is not a function');
if (typeof invoiceSchema !== 'object' || invoiceSchema === null) throw new Error('invoiceSchema is not an object');
if (typeof invoiceSchema.safeParse !== 'function') throw new Error('invoiceSchema.safeParse is not a function');

const keys = Object.keys(invoiceSchema.shape);
const required = ['vendorName', 'invoiceNumber', 'invoiceDate', 'lineItems', 'subtotal', 'total'];
for (const key of required) {
  if (!keys.includes(key)) throw new Error('invoiceSchema missing field: ' + key);
}

console.log('CONSUMER TEST PASSED');
console.log('Exports verified: extract, createAnthropicProvider, invoiceSchema');
console.log('invoiceSchema fields:', keys.join(', '));
`;
  writeFileSync(join(tempDir, 'test.mjs'), testScript);

  // 7. Run test
  const output = run('node test.mjs', tempDir);
  console.log(output);

  // 8. Cleanup
  rmSync(tempDir, { recursive: true, force: true });
  console.log('Consumer test temp dir cleaned up.');
  console.log('\nLIB-01 VERIFIED: @docpipe/core is installable and importable outside the monorepo');
  process.exit(0);
} catch (err) {
  console.error('\nCONSUMER TEST FAILED:', err.message);
  if (err.stdout) console.error('stdout:', err.stdout);
  if (err.stderr) console.error('stderr:', err.stderr);
  if (tempDir) {
    try { rmSync(tempDir, { recursive: true, force: true }); } catch {}
  }
  process.exit(1);
}
```

Iterate until the script exits 0 and prints "CONSUMER TEST PASSED".
