**Step 3: Replace the exported `extract()` with the retry wrapper**

```typescript
const SUPPORTED_MIME_TYPES = ['application/pdf', 'image/png', 'image/jpeg'] as const;
const MAX_RETRIES = 2;

export async function extract<T>(options: ExtractOptions<T>): Promise<ExtractionResult<T>> {
  // Guard: unsupported mimeType — throw before any LLM call
  if (!SUPPORTED_MIME_TYPES.includes(options.input.mimeType as (typeof SUPPORTED_MIME_TYPES)[number])) {
    const err: ExtractionError = {
      code: 'UNSUPPORTED_FILE_TYPE',
      message: `Unsupported file type: ${options.input.mimeType}. Supported types: ${SUPPORTED_MIME_TYPES.join(', ')}`,
      retryable: false,
    };
    throw err;
  }

  let lastValidationErrors: string[] = [];

  for (let attempt = 0; attempt <= MAX_RETRIES; attempt++) {
    try {
      return await extractCore(options, lastValidationErrors);
    } catch (e: unknown) {
      // Classify provider HTTP errors — check for status code in error object
      const status = (e as { status?: number })?.status;
      if (status === 401) {
        const err: ExtractionError = { code: 'INVALID_API_KEY', message: 'Invalid API key (401).', retryable: false };
        throw err;
      }
      if (status === 429) {
        const err: ExtractionError = { code: 'RATE_LIMITED', message: 'Rate limited (429). Try again later.', retryable: false };
        throw err;
      }

      // Check if this is a Zod validation error from the AI SDK's schema enforcement
      // The AI SDK throws NoObjectGeneratedError when the response fails schema validation
      // The error message contains the Zod validation details
      const isValidationError =
        (e instanceof Error && e.name === 'NoObjectGeneratedError') ||
        (e instanceof Error && e.message.includes('validation'));

      if (isValidationError && attempt < MAX_RETRIES) {
        // Retry with validation error fed back
        lastValidationErrors = [e instanceof Error ? e.message : String(e)];
        continue;
      }

      if (isValidationError) {
        // Retries exhausted
        const err: ExtractionError = {
          code: 'VALIDATION_FAILED',
          message: 'Extraction output failed schema validation after maximum retries.',
          retryable: false,
          validationErrors: lastValidationErrors,
        };
        throw err;
      }

      // Any other error: extraction failed
      const err: ExtractionError = {
        code: 'EXTRACTION_FAILED',
        message: `Extraction failed: ${e instanceof Error ? e.message : String(e)}`,
        retryable: true,
      };
      throw err;
    }
  }

  // Should never reach here (loop always throws or returns)
  throw new Error('extract(): unreachable code after retry loop');
}
```
