**Schema definition (exact field names and types):**

```typescript
import { z } from 'zod';

/**
 * Line item on an invoice.
 */
const lineItemSchema = z.object({
  description: z.string().describe('Product or service description'),
  quantity: z.number().describe('Number of units'),
  unitPrice: z.number().describe('Price per unit in document currency'),
  amount: z.number().describe('quantity × unitPrice total for this line'),
});

/**
 * Invoice extraction schema — built-in template for TMPL-01.
 *
 * All monetary values are numbers (not strings). The LLM is instructed
 * to extract numeric values; currency symbols are stripped.
 *
 * Optional fields (dueDate, taxAmount, taxRate) use .nullable() because
 * not all invoices include them.
 */
export const invoiceSchema = z.object({
  vendorName: z.string().describe('Company or individual name issuing the invoice'),
  vendorAddress: z.string().nullable().describe('Full vendor address or null if not present'),
  invoiceNumber: z.string().describe('Invoice identifier/reference number'),
  invoiceDate: z.string().describe('Invoice issue date as printed (e.g., "2024-01-15" or "January 15, 2024")'),
  dueDate: z.string().nullable().describe('Payment due date as printed, or null if not present'),
  lineItems: z.array(lineItemSchema).describe('All billable line items on the invoice'),
  subtotal: z.number().describe('Sum of all line items before tax'),
  taxAmount: z.number().nullable().describe('Tax amount in document currency, or null if not shown'),
  taxRate: z.number().nullable().describe('Tax rate as a decimal (e.g., 0.08 for 8%), or null if not shown'),
  total: z.number().describe('Final total amount due'),
});

/**
 * TypeScript type inferred from invoiceSchema.
 * Use as the generic parameter: ExtractionResult<InvoiceData>
 */
export type InvoiceData = z.infer<typeof invoiceSchema>;
```
