**Create `packages/core/src/pdf-router.ts`:**

```typescript
import { extractText } from 'unpdf';

/**
 * PDF document type — result of inspecting text layer presence.
 * - 'text-layer': PDF has extractable text (selectable, copy-pasteable)
 * - 'image-only': PDF is a scanned image — no extractable text layer
 */
export type PdfType = 'text-layer' | 'image-only';

/**
 * Minimum number of non-whitespace characters required to classify a PDF
 * as having a usable text layer. Calibrated against real scanned PDFs:
 * - Scanned PDFs: 0-5 chars (occasional OCR artifacts from pdfjs)
 * - Text PDFs: typically 100-10000+ chars per page
 * - Threshold of 50 provides a wide safety margin
 */
const TEXT_LAYER_THRESHOLD = 50;

/**
 * Detects whether a PDF has a text layer or is image-only.
 *
 * Uses unpdf to attempt text extraction. Counts non-whitespace characters.
 * If the count is below TEXT_LAYER_THRESHOLD, the PDF is classified as image-only.
 *
 * Both types are handled by Claude's native `document` content block —
 * this detection is for annotation and future routing optimization only.
 *
 * @param pdfBuffer - Raw PDF file as a Node.js Buffer
 * @returns 'text-layer' if PDF has extractable text, 'image-only' otherwise
 */
export async function detectPdfType(pdfBuffer: Buffer): Promise<PdfType> {
  try {
    const uint8 = new Uint8Array(pdfBuffer);
    const { text } = await extractText(uint8);
    const nonWhitespaceCount = (text ?? '').replace(/\s/g, '').length;
    return nonWhitespaceCount >= TEXT_LAYER_THRESHOLD ? 'text-layer' : 'image-only';
  } catch {
    // If text extraction fails (corrupt PDF, password-protected, etc.),
    // fall back to image-only — Claude will still attempt extraction via document block
    return 'image-only';
  }
}
```

import { vi, describe, it, expect, beforeEach } from 'vitest';
import { detectPdfType } from './pdf-router.js';

vi.mock('unpdf', () => ({
  extractText: vi.fn(),
}));

import { extractText } from 'unpdf';
const mockExtractText = vi.mocked(extractText);

beforeEach(() => {
  vi.resetAllMocks();
});

describe('detectPdfType', () => {
  it('returns text-layer when extracted text has 200+ non-whitespace chars', async () => {
    mockExtractText.mockResolvedValueOnce({ text: 'a'.repeat(200), totalPages: 1 });
    const result = await detectPdfType(Buffer.from('fake-pdf'));
    expect(result).toBe('text-layer');
  });

  it('returns image-only when extracted text is empty string', async () => {
    mockExtractText.mockResolvedValueOnce({ text: '', totalPages: 1 });
    const result = await detectPdfType(Buffer.from('fake-pdf'));
    expect(result).toBe('image-only');
  });

  it('returns image-only when extracted text is only whitespace', async () => {
    mockExtractText.mockResolvedValueOnce({ text: '   \n  \t  ', totalPages: 1 });
    const result = await detectPdfType(Buffer.from('fake-pdf'));
    expect(result).toBe('image-only');
  });

  it('returns image-only when non-whitespace count is exactly 49 (below threshold)', async () => {
    const text49 = 'a'.repeat(49) + '   ';
    mockExtractText.mockResolvedValueOnce({ text: text49, totalPages: 1 });
    const result = await detectPdfType(Buffer.from('fake-pdf'));
    expect(result).toBe('image-only');
  });

  it('returns text-layer when non-whitespace count is exactly 50 (at threshold)', async () => {
    const text50 = 'a'.repeat(50);
    mockExtractText.mockResolvedValueOnce({ text: text50, totalPages: 1 });
    const result = await detectPdfType(Buffer.from('fake-pdf'));
    expect(result).toBe('text-layer');
  });

  it('returns image-only when extractText throws (corrupt or password-protected PDF)', async () => {
    mockExtractText.mockRejectedValueOnce(new Error('PDF parsing failed'));
    const result = await detectPdfType(Buffer.from('fake-pdf'));
    expect(result).toBe('image-only');
  });
});
