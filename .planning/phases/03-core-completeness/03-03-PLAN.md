**Part B: Create `packages/core/src/validators/` directory and `invoice-validators.ts`**

```typescript
import type { InvoiceData } from '../templates/invoice.js';
import type { ExtractionWarning } from '../types.js';

/**
 * Tolerance for floating-point comparison in monetary sums.
 * Line items summing within 1 cent of subtotal are considered matching.
 */
const CURRENCY_TOLERANCE = 0.01;

/**
 * Business-rule validators for invoice extraction results.
 * Returns an array of warnings (empty array = no anomalies found).
 * Never throws — all errors produce a warning, not an exception.
 *
 * Rules validated:
 * 1. ZERO_TOTAL: total is 0 or negative
 * 2. FUTURE_DATE: invoiceDate parses to a date in the future
 * 3. LINE_ITEMS_SUBTOTAL_MISMATCH: sum of lineItem.amount differs from subtotal by more than 1 cent
 */
export function validateInvoice(data: InvoiceData): ExtractionWarning[] {
  const warnings: ExtractionWarning[] = [];

  // Rule 1: Zero or negative total
  if (data.total <= 0) {
    warnings.push({
      code: 'ZERO_TOTAL',
      message: `Invoice total is ${data.total} — expected a positive amount. This may indicate an extraction error.`,
      field: 'total',
    });
  }

  // Rule 2: Future invoice date
  // Only check if the date string parses to a valid Date
  const parsedDate = new Date(data.invoiceDate);
  if (!isNaN(parsedDate.getTime()) && parsedDate > new Date()) {
    warnings.push({
      code: 'FUTURE_DATE',
      message: `Invoice date "${data.invoiceDate}" is in the future. This may indicate a data entry error or incorrect extraction.`,
      field: 'invoiceDate',
    });
  }

  // Rule 3: Line items do not sum to subtotal
  // Only check if subtotal is present (not null)
  if (data.subtotal !== null && data.subtotal !== undefined && data.lineItems.length > 0) {
    const lineItemSum = data.lineItems.reduce((sum, item) => sum + item.amount, 0);
    if (Math.abs(lineItemSum - data.subtotal) > CURRENCY_TOLERANCE) {
      warnings.push({
        code: 'LINE_ITEMS_SUBTOTAL_MISMATCH',
        message: `Line items sum to ${lineItemSum.toFixed(2)} but subtotal is ${data.subtotal.toFixed(2)}. Difference: ${Math.abs(lineItemSum - data.subtotal).toFixed(2)}.`,
        field: 'subtotal',
      });
    }
  }

  return warnings;
}
```

2. After the successful `extractCore()` call inside the retry loop (before returning), run validators:

```typescript
// After extractCore returns successfully:
const extractionResult = await extractCore(options, lastValidationErrors);

// Run business-rule validators if provided
let warnings: ExtractionWarning[] | undefined;
if (options.validators && options.validators.length > 0) {
  const allWarnings = options.validators.flatMap((validator) =>
    validator(extractionResult.data),
  );
  warnings = allWarnings.length > 0 ? allWarnings : undefined;
}

return {
  ...extractionResult,
  ...(warnings !== undefined ? { warnings } : {}),
};
```

Note: The validators run AFTER the extraction succeeds. They do not trigger retries — only Zod schema validation failures cause retries. If a validator throws, let the error propagate as an EXTRACTION_FAILED error (caught by the existing error handler).
